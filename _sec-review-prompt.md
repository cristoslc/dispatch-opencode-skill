+#!/usr/bin/env bash
+# Rendered dispatch script — pr-work
+#
+# Creates a draft PR, then works in the worktree with the PR as chronicle.
+# Writes .lock on start, FINAL_OUTPUT.md on completion, deletes .lock on exit.
+set -uo pipefail
+
+TASK_DIR={{ task_dir | shellquote }}
+TASK_ID={{ task_id | shellquote }}
+GENERATED_AT={{ generated_at | shellquote }}
+CWD={{ cwd | shellquote }}
+MODEL={{ model | shellquote }}
+AGENT={{ agent | shellquote }}
+BRANCH={{ branch | shellquote }}
+PR_TITLE={{ pr_title | shellquote }}
+
+echo "PID=$$" > "$TASK_DIR/.lock"
+
+# Detect serve vs local mode (same pattern as single-file-fix)
+if [ -n "${OPENCODE_SERVER_URL:-}" ]; then
+  ATTACH_ARGS=(--attach "$OPENCODE_SERVER_URL" --password "$OPENCODE_SERVER_PASSWORD")
+else
+  unset OPENCODE_SERVER_PASSWORD
+  ATTACH_ARGS=(--dir "$CWD")
+fi
+export OPENCODE_DISABLE_AUTOCOMPACT=true
+export OPENCODE_DISABLE_AUTOUPDATE=true
+
+echo "[dispatch-opencode] task_id=$TASK_ID generated=$GENERATED_AT"
+echo "[dispatch-opencode] cwd=$CWD"
+echo "[dispatch-opencode] model=$MODEL agent=$AGENT"
+echo "[dispatch-opencode] branch=$BRANCH pr_title=$PR_TITLE"
+
+# Push branch to remote (ensure it exists for PR creation)
+git push origin "HEAD:$BRANCH" 2>&1 || echo "[dispatch-opencode] warn: git push failed (branch may already exist upstream)"
+
+# Create draft PR with the prompt as the body
+PR_URL=""
+if command -v gh &>/dev/null; then
+  PR_URL=$(gh pr create --draft --title "$PR_TITLE" --body-file "$TASK_DIR/prompt.md" 2>/dev/null) || {
+    echo "[dispatch-opencode] warn: gh pr create failed (continue without PR)" >&2
+    PR_URL=""
+  }
+  echo "[dispatch-opencode] pr_url=$PR_URL"
+else
+  echo "[dispatch-opencode] warn: gh not on PATH; skipping PR creation" >&2
+fi
+export PR_URL
+
+TIMEOUT_BIN="$(command -v gtimeout 2>/dev/null || command -v timeout 2>/dev/null || true)"
+if [ -n "$TIMEOUT_BIN" ]; then
+  TIMEOUT_PREFIX=("$TIMEOUT_BIN" "600")

+  echo "[dispatch-opencode] warn: no timeout(1)/gtimeout on PATH; running without timeout" >&2
+  TIMEOUT_PREFIX=()
+fi
+

+  "${ATTACH_ARGS[@]}" \

+  --agent "$AGENT" \
